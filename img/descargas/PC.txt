 //Una opción para ordenar por un atributo numérico.  (2puntos) (nagito)
    public int hallaMenorPC1(int i, int indMax){
        double menor = arregloAlumno[i].getPc1(); int indMenor =i;
        for( int j =i; j<=indMax; j++){
            if(arregloAlumno[j].getPc1()< menor){
                menor = arregloAlumno[j].getPc1(); indMenor = j;
            }
        }
        return indMenor;
    }
    
    public void ordenarPorPc1(){
        int indMax=arregloAlumno.length-1;    
        int indMenor;
        Alumno aux;
        for(int i = 0; i <= indMax; i++){
            indMenor = hallaMenorPC1(i, indMax);
            aux = arregloAlumno[indMenor];
            
            arregloAlumno[indMenor] =arregloAlumno[i];
            arregloAlumno[i]=aux;
            
        }    
    }
